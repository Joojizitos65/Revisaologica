CREATE TABLE compradores (
  id INT PRIMARY KEY,
  nome VARCHAR(50)
);

CREATE TABLE vendedores (
  id INT PRIMARY KEY,
  nome VARCHAR(50)
);

CREATE TABLE gastos_mensais (
  id INT PRIMARY KEY,
  mes VARCHAR(10),
  comprador_id INT,
  vendedor_id INT,
  valor DECIMAL(10, 2),
  FOREIGN KEY (comprador_id) REFERENCES compradores(id),
  FOREIGN KEY (vendedor_id) REFERENCES vendedores(id)
);
